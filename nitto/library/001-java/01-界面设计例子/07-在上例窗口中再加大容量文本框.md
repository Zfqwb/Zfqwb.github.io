## 在上例窗口中再加大容量文本框



```java

import  java.awt.*;

import javax.swing.*;
public class J16  extends  J15 {
public J16(){
	JPanel  p17=new JPanel();
	setLayout(new GridLayout(7,1));	
	JLabel jLabel1=new JLabel("个人简介：");
	jLabel1.setPreferredSize(new Dimension(75,50));
	JTextArea ta4=new JTextArea("",2,20);
	ta4.setFont(new Font("Serif", Font.PLAIN, 12));
	ta4.setTabSize(10);
	ta4.setLineWrap(true);
	ta4.setWrapStyleWord(true);
	add(p17);
	JScrollPane gbp=new JScrollPane(ta4);
	p17.add(jLabel1);p17.add(gbp);
}	
public static void main(String[] args){
	J16 d=new J16();
	d.setVisible(true);
}	

}
```
