## 在上例窗口中再加复选框并加监听事件（采用内部类方式）

```java

import  java.awt.*;
import java.awt.event.*;

import javax.swing.*;
public class J14 extends J13 {
    private JLabel labe1;
	private JCheckBox b, i, s, g;
	JPanel p15;
		public J14() {				
		p15 = new JPanel();	
		labe1 = new JLabel("爱好：");//创建标签对象
		b = new JCheckBox("唱歌");//创建复选框
		i = new JCheckBox("跳舞");//创建复选框
		s = new JCheckBox("绘画");//创建复选框
		g = new JCheckBox("运动");//创建复选框
		b.setFont(new Font("Serif", Font.PLAIN, 14));// 设置复选框选项标签字体
		i.setFont(new Font("Serif", Font.PLAIN, 14));
		s.setFont(new Font("Serif", Font.PLAIN, 14));
		g.setFont(new Font("Serif", Font.PLAIN, 14));
		CBHandler  h=new CBHandler();
		b.addItemListener(h);
		i.addItemListener(h);
		s.addItemListener(h);
		g.addItemListener(h);
		setLayout(new GridLayout(5,1));
		p15.add(labe1);
		p15.add(b);
		p15.add(i);
		p15.add(s);
		p15.add(g);
		add(p15);
		}
		class CBHandler  implements  ItemListener{
			private int vb=Font.PLAIN;
			private int vi=Font.PLAIN;
			private int vs=Font.PLAIN;
			private int vg=Font.PLAIN;
		   
			public void itemStateChanged(ItemEvent  e){
				if(e.getSource()==b)  vb=b.isSelected()?Font.BOLD:Font.PLAIN;
				if(e.getSource()==b)  vi=b.isSelected()?Font.BOLD:Font.PLAIN;
				if(e.getSource()==b)  vs=b.isSelected()?Font.BOLD:Font.PLAIN;
				if(e.getSource()==b)  vg=b.isSelected()?Font.BOLD:Font.PLAIN;
				labe1.setFont(new Font("serif",vb+vi+vs+vg,15));
			}
		   }
		   
		    public static void main(String args[]){J14 d= new J14();d.setVisible(true);	
		    }
		    }


```
