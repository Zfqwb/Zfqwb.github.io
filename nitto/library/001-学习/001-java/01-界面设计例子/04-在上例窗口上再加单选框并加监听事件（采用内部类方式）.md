## 在上例窗口上再加单选框并加监听事件（采用内部类方式）



```java
import javax.swing.*;
import  java.awt.*;
import java.awt.event.*;
public class J13 extends J12{ 
	private  JLabel j1,j2;
	private JRadioButton  jb1,jb2;
    public J13() {
    	JPanel  p14 = new JPanel();
        JLabel  j1 = new JLabel("性别："); 
        JLabel  j2 = new JLabel(new ImageIcon("src/图片/11.gif"));       
        
        JRadioButton  jb1 = new JRadioButton("男",true);//选项按钮1
        JRadioButton  jb2 = new JRadioButton("女");     //选项按钮2
        ButtonGroup  buttonGroup = new ButtonGroup();   //单选按钮组
        buttonGroup.add(jb1);buttonGroup.add(jb2);      //添加选项按钮到组中
        setLayout(new GridLayout(4,1));
        add(p14);p14.add(j1);  p14.add(jb1);   p14.add(jb2);  p14.add(j2);
        rHandler  h=new rHandler();
        jb1.addItemListener(h);
        jb2.addItemListener(h);
      }
   class rHandler  implements  ItemListener{
	public void itemStateChanged(ItemEvent  event){
		if(jb1.isSelected())  j2.setIcon(new ImageIcon("src/图片/11.gif"));
		else if(jb2.isSelected())  j2.setIcon(new ImageIcon("src/图片/12.gif"));
		else j2.setBackground(Color.white);
	}
   }
   
    public static void main(String args[]){J13 d= new J13();d.setVisible(true);	 }
}
```
